	placement = root

	flavor = dreyfus.1.f

		has_dlc_feature = voice_of_the_people_content
		uses_fra_characters = yes		##Converter
		has_technology_researched = political_agitation

			OR = {
				has_variable = recently_lost_war
				AND = {
					has_variable = recently_had_war
					NOT = {
						has_variable = recently_won_war
					}
				}
			}
			any_state = {
				state_region = { 
					is_homeland = global_var:fra_culture		##Converter
				}
				NOT = {
					OR = {
						state_region = s:STATE_SAVOY
						AND = {
							state_region = s:STATE_PROVENCE
							state_region = {
								any_scope_state = {
									owner = root
								}
							}
							is_split_state = yes
						}
					}
					owner = root
				}
			}
		}
		ig:ig_armed_forces = {
			is_marginal = no
		}
		NOT = {
			has_law = law_type:law_multicultural
		}
		NOT = { 
			has_variable = dreyfus_convicted
		}
		any_scope_state = {
        	state_region = { 
        		is_homeland = global_var:fra_culture		##Converter
        	}
        	any_scope_pop = {
        		religion = rel:jewish
        	}
		}
		any_country = {
			has_diplomatic_pact = {
				who = root
				type = rivalry
			}
		}
	}

		##Converter

			limit = {
				any_scope_pop = { culture = cu:ashkenazi }
			}
			cu:ashkenazi = { save_scope_as = dreyfus_culture }
		}

			limit = {
				any_scope_pop = { culture = { has_discrimination_trait = jewish_group } }
			}
			random_scope_pop = {
				limit = { culture = { has_discrimination_trait = jewish_group } }
				culture = { save_scope_as = dreyfus_culture }
			}
		}
		else_if = {
			limit = {
				any_scope_pop = { religion = { has_discrimination_trait = jewish } }
			}
			random_scope_pop = {
				limit = { religion = { has_discrimination_trait = jewish } }
				culture = { save_scope_as = dreyfus_culture }
			}
		}
		else = {
			cu:ashkenazi = { save_scope_as = dreyfus_culture }
		}
		converter_rule_63_effect = yes
		if = {
			limit = { has_variable = rule_63_effect }

				template = fem_dreyfus_character_template

dreyfus.4 = {
	type = country_event
	placement = root
	title = dreyfus.4.t
	desc = dreyfus.4.d
	flavor = dreyfus.4.f

	duration = 3

	dlc = dlc004

	event_image = {
		video = "votp_agitators_sneaking"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	trigger = {
		OR = {
			any_scope_character = {
				has_variable = emile_zola_var
				is_busy = no
			}
			any_interest_group = {
				ig_is_dreyfusard = yes
				leader = {
					is_busy = no
				}
			}
		}
		NOT = { has_variable = zola_asphyxiation }
		var:rennes_retrial_var >= 35
	}

	immediate = {
		ig:ig_armed_forces = {
			save_scope_as = dreyfus_armed_forces_scope
		}
		random_character_in_void = {
			limit = {
				has_variable = alfred_dreyfus_var
			}
			save_scope_as = alfred_dreyfus_scope
		}
		if = {
			limit = {
				any_scope_character = {
					has_variable = emile_zola_var
				}
			}
			random_scope_character = {
				limit = {
					has_variable = emile_zola_var
					is_busy = no
				}
				save_scope_as = emile_zola_scope
				interest_group = {
					save_scope_as = zola_asphyxiation_ig_pro
				}
			}
		}
		else_if = {
			random_interest_group = {
				limit = {
					ig_is_dreyfusard = yes
					leader = {
						is_busy = no
					}
				}
				leader = {
					save_scope_as = zola_asphyxiation_leader
				}
				save_scope_as = zola_asphyxiation_ig_pro
			}
		}
		set_variable = zola_asphyxiation
		if = {
			limit = {
				any_interest_group = {
					ig_is_anti_dreyfusard = yes
					NOT = { is_interest_group_type = ig_armed_forces }
				}
			}
			random_interest_group = {
				limit = {
					ig_is_anti_dreyfusard = yes
					NOT = { is_interest_group_type = ig_armed_forces }
				}
				save_scope_as = zola_asphyxiation_ig_anti
			}
		}
	}

	option = {
		name = dreyfus.4.a
		default_option = yes
		scope:zola_asphyxiation_leader = {
			kill_character = yes
		}
		custom_tooltip = {
			text = dreyfus_case_strength_decrease_by_1
			change_variable = {
				name = dreyfus_case_var
				add = -1
			}
		}
	}
	option = {
		name = dreyfus.4.b
		scope:zola_asphyxiation_leader = {
			kill_character = yes
		}
		scope:zola_asphyxiation_ig_pro = {
			add_modifier = {
				name = zola_state_funeral_positive
				months = normal_modifier_time
			}
		}
		scope:dreyfus_armed_forces_scope = {
			add_modifier = {
				name = zola_state_funeral_negative
				months = normal_modifier_time
			}
		}
		if = {
			limit = {
				exists = scope:zola_asphyxiation_ig_anti
			}
			scope:zola_asphyxiation_ig_anti = {
				add_modifier = {
					name = zola_state_funeral_negative
					months = normal_modifier_time
				}
			}
		}
		add_modifier = {
			name = zola_state_funeral_expenses
			months = short_modifier_time
			multiplier = money_amount_multiplier_medium
		}
	}
	option = {
		name = dreyfus.4.c
		scope:zola_asphyxiation_leader = {
			kill_character = yes
		}
		custom_tooltip = {
			text = dreyfus_case_strength_increase_by_1
			change_variable = {
				name = dreyfus_case_var
				add = 1
			}
		}
	}
}
