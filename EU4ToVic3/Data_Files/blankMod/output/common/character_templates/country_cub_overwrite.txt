	}

			uses_cuba_characters = yes
			should_have_male_characters = yes
			NOT = {
				is_target_in_global_variable_list = {
					name = converter_unavailable_unique_characters
					target = flag:converter_antonio_maceo
				}
			}
			OR = {
				is_subject = no
				AND = {
					is_subject = yes

				}
			}
			has_dlc_feature = ip4_content
		}
		role = general
		earliest_usage_date = 1874.1.1
		latest_usage_date = 1895.1.1
		chance = 50
	}
	on_created = {

		add_to_global_variable_list = {
			name = converter_unavailable_unique_characters
			target = flag:converter_antonio_maceo
		}
		fix_variable_error = { variable = converter_antonio_maceo }
		if = {
			limit = {
				home_country ?= {
					is_subject = no
				}
			}
			set_ideology = ideology:ideology_radical # Dios, la Razón y la Virtud
		}
	}
}
