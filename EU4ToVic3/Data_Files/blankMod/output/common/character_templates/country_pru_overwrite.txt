PRU_otto_von_bismarck = {
	first_name = Otto
	last_name = von_Bismarck
	historical = yes
	ig_leader = yes
	interest_group = ig_landowners
	female = no
	birth_date = 1815.4.1
	dna = dna_otto_von_bismarck
	ideology = ideology_royalist
	traits = {
		direct
		cautious
	}
	on_created = {
		converter_standard_template_effect = { TARGET = cu:north_german }
		add_to_global_variable_list = {
			name = converter_unavailable_unique_characters
			target = flag:converter_otto_von_bismarck
		}
		fix_variable_error = { variable = converter_otto_von_bismarck }
	}
}
