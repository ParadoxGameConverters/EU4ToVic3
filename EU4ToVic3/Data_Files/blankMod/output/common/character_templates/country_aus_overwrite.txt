				}
			}

        latest_usage_date = 1900.1.1
        chance = 100

	on_created = {
		converter_standard_template_effect = { TARGET = cu:czech }

			name = converter_unavailable_unique_characters
			target = flag:converter_emil_von_skoda
		}
		fix_variable_error = { variable = converter_emil_von_skoda }
	}
}
