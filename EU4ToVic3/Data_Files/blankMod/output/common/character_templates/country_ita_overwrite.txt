ITA_giovanni_agnelli = { # Giovanni Agnelli, Fiat
	first_name = Giovanni
	last_name = Agnelli
	historical = yes
	birth_date = 1866.8.13
	dna = dna_giovanni_agnelli
	culture = cu:north_italian
	interest_group = ig_industrialists
	ideology = ideology_moderate
	traits = {
		ambitious
		engineer
		basic_entrepreneur
	}
	executive_usage = {
		country_trigger = {
			should_have_male_characters = yes
			NOT = {
				is_target_in_global_variable_list = {
					name = converter_unavailable_unique_characters
					target = flag:converter_giovanni_agnelli
				}
			}
			has_dlc_feature = mp1_content
		}
		company_trigger = {
			is_company_type = company_type:company_fiat
		}
		earliest_usage_date = 1899.1.1
		latest_usage_date = 1936.1.1
		chance = 100
	}
	on_created = {
		converter_standard_template_effect = { TARGET = cu:north_italian }
		add_to_global_variable_list = {
			name = converter_unavailable_unique_characters
			target = flag:converter_giovanni_agnelli
		}
		fix_variable_error = { variable = converter_giovanni_agnelli }
	}
}
