converter_has_shinto_faith = {
	OR = {
		country_has_state_religion = rel:shinto
		religion = { is_target_in_global_variable_list = { name = global_shinto_branch_religions target = this } }
	}
}

	OR = {
		var:tc_majority_religion = rel:shinto
		var:tc_majority_religion = { is_target_in_global_variable_list = { name = global_shinto_branch_religions target = this } }
	}
}
