is_settler_colonial_country = {
	coa_republic_trigger = yes
	trigger_if = {
		limit = {
			OR = {
				country_is_in_north_america = yes
				country_is_in_central_america = yes
				country_is_in_south_america = yes
			}
		}
		NOT = { any_primary_culture = { has_discrimination_trait = indigenous_american_heritage } }
	}
	trigger_else_if = {
		limit = {
			exists = capital
			capital = { region = sr:region_oceania }
		}
		NOT = { any_primary_culture = { has_discrimination_trait = indigenous_oceanic_heritage } }
	}
	trigger_else_if = {
		limit = { country_is_in_africa = yes }
		any_primary_culture = { has_discrimination_trait = african_settler }
		NOT = { any_primary_culture = { has_discrimination_trait = african_heritage } }
	}
	trigger_else = { always = no }
}

		}
		this = c:USA
	}
	trigger_else = {
		is_target_in_global_variable_list = { name = global_new_england_yankee_countries target = this }
	}
}

	is_target_in_global_variable_list = { name = global_ber_countries target = this }
	is_subject = no
	any_scope_state = {
		OR = {
			state_region = s:STATE_ORAN
			state_region = s:STATE_ALGIERS

		country_is_in_africa = no
