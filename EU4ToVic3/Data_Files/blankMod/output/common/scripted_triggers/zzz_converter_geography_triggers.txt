converter_is_converter_confederation_culture = {
	trigger_if = {
		limit = { has_game_rule = converter_confederation_cultures_all }
		always = yes
	}
	trigger_else_if = {
		limit = { has_game_rule = converter_confederation_cultures_any }
		shares_trait_with_any_primary_culture = prev.owner
	}
	trigger_else_if = {
		limit = { has_game_rule = converter_confederation_cultures_heritage }
		shares_heritage_trait_with_any_primary_culture = prev.owner
	}
	trigger_else_if = {
		limit = { has_game_rule = converter_confederation_cultures_language }
		shares_non_heritage_trait_with_any_primary_culture = prev.owner
	}
	trigger_else_if = {
		limit = { has_game_rule = converter_confederation_cultures_heritage_language }
		shares_heritage_and_other_trait_with_any_primary_culture = prev.owner
	}
	trigger_else = { always = no }
}

converter_should_add_confederation_cultures = {
	trigger_if = {
		limit = { is_subject = yes }
		OR = {
			top_overlord = {
				capital = {
					OR = {
						state_is_in_americas = yes
						region = sr:region_oceania
					}
				}
			}
			this = c:CAN
			this = c:USA
			this = c:CAL
			this = c:MEX
			this = c:GCO
			this = c:PEU
			this = c:BRZ
			this = c:ARG
			this = c:AST
			this = c:SAF
		}
	}
	trigger_else = { always = yes }
}

is_settler_colonial_country = {
	trigger_if = {
		limit = {
			OR = {
				country_is_in_north_america = yes
				country_is_in_central_america = yes
				country_is_in_south_america = yes
			}
		}
		NOT = { any_primary_culture = { has_discrimination_trait = indigenous_american_heritage } }
	}
	trigger_else_if = {
		limit = {
			exists = capital
			capital = { region = sr:region_oceania }
		}
		NOT = { any_primary_culture = { has_discrimination_trait = indigenous_oceanic_heritage } }
	}
	trigger_else_if = {
		limit = { country_is_in_africa = yes }
		any_primary_culture = { has_discrimination_trait = african_settler }
		NOT = { any_primary_culture = { has_discrimination_trait = african_heritage } }
	}
	trigger_else = { always = no }
}
