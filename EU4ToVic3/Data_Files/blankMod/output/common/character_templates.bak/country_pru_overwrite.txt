REPLACE:PRU_alfred_krupp = { # Alfred Krupp, Fried. Krupp AG
	first_name = Alfred
	last_name = Krupp
	historical = yes
	birth_date = 1812.4.26
	dna = dna_alfred_krupp
	culture = cu:north_german
	interest_group = ig_industrialists
	ideology = ideology_authoritarian
	traits = {
		bigoted
		basic_entrepreneur
	}
	executive_usage = {
		country_trigger = {
			should_have_male_characters = yes
			NOT = {
				is_target_in_global_variable_list = {
					name = converter_unavailable_unique_characters
					target = flag:converter_alfred_krupp
				}
			}
			has_dlc_feature = mp1_content
		}
		company_trigger = {
			is_company_type = company_type:company_krupp
		}
		earliest_usage_date = 1836.1.1
		latest_usage_date = 1887.1.1
		chance = 100
	}
	on_created = {
		converter_standard_template_effect = { TARGET = cu:north_german }
		add_to_global_variable_list = {
			name = converter_unavailable_unique_characters
			target = flag:converter_alfred_krupp
		}
		fix_variable_error = { variable = converter_alfred_krupp }
	}
}
